## Usage

```tsx
import { PlusIcon } from "lucide-react"

import {
  IconHoverButton,
  IconHoverButtonIcon,
  IconHoverButtonText,
} from "@/components/ui/icon-hover-button"
```

```tsx
<IconHoverButton>
  <IconHoverButtonIcon>
    <PlusIcon />
  </IconHoverButtonIcon>
  <IconHoverButtonText>Button</IconHoverButtonText>
</IconHoverButton>
```
